    function removeClass(element, className) {
  !!/\s/.test(className) ? process.env.NODE_ENV !== 'production' ? invariant(false, 'CSSCore.removeClass takes only a single class
 name. "%s" contains ' + 'multiple classes.', className) : invariant(false) : undefined;

  if (className) {
    if (element.classList) {
      element.classList.remove(className);
    } else if (CSSCore.hasClass(element, className)) {

      .replace(/^\s*|\s*$/g, ''); // trim the ends
    }
  }
  return element;
}

    function fiveArgumentPooler(a1, a2, a3, a4, a5) {
  var Klass = this;
  if (Klass.instancePool.length) {
    var instance = Klass.instancePool.pop();
    Klass.call(instance, a1, a2, a3, a4, a5);
    return instance;
  } else {
    return new Klass(a1, a2, a3, a4, a5);
  }
